#+-----------------Assignment----------------+#

#+-----------------And----------------+#

a := 1
d := 1
c := 2

if (a > d AND c > d)
{
    write("a es mas grande que d y c es mas grande que d")
}


#+-----------------Or----------------+#

if (a > d OR c > d)
{
    write("a es mas grande que d o c es mas grande que d")
}


#+-----------------Not----------------+#

if (NOT a > d)
{
    write("a no es mas grande que d")
}

#+-----------------Read----------------+#

read(base)

#+-----------------Write----------------+#

write("ewr")
write(var1)

#+-----------------If/Else----------------+#

if (a > d)
{
    write("a es mas grande que d")
}
else
{
    write("a es mas chico o igual a d")
}


#+-----------------While----------------+#


while (a > d)
{
    write("a es mas grande que d")
    a := a + 1
}


#+-----------------Long Function----------------+#


if (long([1+2, 3*4, x, y+z]) > 3) 
{
    write("lista es mas grande que 3")
}
